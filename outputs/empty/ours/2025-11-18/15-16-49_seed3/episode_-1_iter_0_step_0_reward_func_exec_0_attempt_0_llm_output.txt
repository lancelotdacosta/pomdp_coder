def reward_func(state, action, next_state):
    """
    Args:
        state (MinigridState): the state of the environment
        action (int): the action to be executed
        next_state (MinigridState): the next state of the environment
    Returns:
        reward (float): the reward of that state
        done (bool): whether the episode is done
    """
    
    def is_goal_reached(state):
        # Check if the agent is on top of the goal object (10)
        return state.grid[state.agent_pos] == ObjectTypes.goal
    
    def is_terminal(state):
        # Check if the agent has reached a terminal state
        return is_goal_reached(state)
    
    # Determine the reward based on the next state
    if is_goal_reached(next_state):
        # If the agent reaches the goal, reward is 1 and the episode is done
        return 1, True
    else:
        # If the agent does not reach the goal, reward is 0 and the episode continues
        return 0, False
