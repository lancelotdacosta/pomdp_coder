def reward_func(state, action, next_state):
    """
    Args:
        state (MinigridState): the state of the environment
        action (int): the action to be executed
        next_state (MinigridState): the next state of the environment
    Returns:
        reward (float): the reward of that state
        done (bool): whether the episode is done
    """

    # Check if the agent has moved into a goal cell
    if next_state.grid[next_state.agent_pos] == ObjectTypes.goal:
        return 1.0, True

    # Check if the agent has picked up an object
    if state.carrying is None and next_state.carrying is not None:
        return 0.5, False

    # Check if the agent has dropped an object
    if state.carrying is not None and next_state.carrying is None:
        return 0.5, False

    # Check if the agent has toggled an object (e.g., opened a door)
    if action == Actions.toggle:
        front_pos = next_state.front_pos
        if state.grid[front_pos] != next_state.grid[front_pos]:
            return 0.5, False

    # Check if the agent has moved into a wall or lava cell
    if next_state.grid[next_state.agent_pos] in [ObjectTypes.wall, ObjectTypes.lava]:
        return -1.0, True

    # Check if the agent has completed the task (if any specific task is defined)
    if action == Actions.done:
        return 0.0, True

    # Default reward for moving around
    return 0.0, False
