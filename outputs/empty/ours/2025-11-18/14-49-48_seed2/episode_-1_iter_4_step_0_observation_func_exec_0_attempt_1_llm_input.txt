        return self.grid.shape[1]

        DO NOT modify this function.

            topY = self.agent_pos[1] - view_size // 2
        # Facing down
        elif self.agent_dir == 1:
            topX = self.agent_pos[0] - view_size // 2
            topY = self.agent_pos[1]
        # Facing left
        elif self.agent_dir == 2:
            topX = self.agent_pos[0] - view_size + 1
            topY = self.agent_pos[1] - view_size // 2
        # Facing up
        elif self.agent_dir == 3:
            topX = self.agent_pos[0] - view_size // 2
            topY = self.agent_pos[1] - view_size + 1

        gy1 = min(topY + view_size, self.grid.shape[1])

        ]

        for _ in range(self.agent_dir + 1):

            fov = np.rot90(fov.T, k=1).T

        agent_pos = (self.grid.shape[0] // 2, self.grid.shape[1] - 1)
        self.grid[agent_pos] = ObjectTypes.agent

        return fov

        empty_obs (MinigridObservation): an empty observation that needs to be filled and returned

    def apply_action(state, action):
        """
        Apply the given action to the state and return the new state.
        """
        if action == Actions.forward:
            next_pos = np.array(state.agent_pos) + np.array(DIR_TO_VEC[state.agent_dir])
            if (0 <= next_pos[0] < state.width and 0 <= next_pos[1] < state.height and 
                state.grid[next_pos[0], next_pos[1]] != ObjectTypes.wall):
                state.agent_pos = tuple(next_pos)
        elif action == Actions.left:
            state.agent_dir = (state.agent_dir - 1) % 4
        elif action == Actions.right:
            state.agent_dir = (state.agent_dir + 1) % 4
        # Note: Actions pickup, drop, toggle, and done are not implemented in this function for simplicity.
        return state

    def get_field_of_view(state):

        Get the field of view based on the agent's current position and direction.
        """

        fov = state.get_field_of_view(view_size)

    new_state = apply_action(state, action)

        agent_pos=new_state.agent_pos,
        agent_dir=new_state.agent_dir,
        carrying=new_state.carrying
    )

    return obs

Certainly! Let's start by explaining the distribution of observations conditioned on a state.

### Explanation of Observation Distribution

The agent's observation is a 3x3 field of view (FoV) centered around the agent's position and oriented according to the agent's direction. The FoV should show what the agent can see immediately in front of it, considering the grid's boundaries and any walls that might obstruct the view. The agent itself is always represented in the FoV, and the FoV should be rotated to match the agent's current direction.

### Issues in the Previous Code

1. **Incorrect Agent Position Update**: The agent's position was being updated incorrectly after the action. Specifically, after moving forward, the agent's position was not being updated correctly in the FoV.
2. **Incorrect FoV Rotation**: The FoV was not being rotated correctly to match the agent's direction.
3. **Incorrect Representation of the Agent in the FoV**: The agent was not being placed in the correct position in the FoV.
