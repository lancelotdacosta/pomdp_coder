    HEAR_LEFT = 0

    NONE = 2

class TigerActions(enum.IntEnum):

    OPEN_LEFT = 0
    OPEN_RIGHT = 1
    LISTEN = 2

@dataclass(frozen=True)
class TigerObservation(Observation):

class TigerState(State):

def initial_func(empty_state: TigerState) -> TigerState:
