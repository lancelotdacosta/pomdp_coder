    OPEN_LEFT = 0

class TigerObservation(Observation):

class TigerState(State):

    """

        state (TigerState): the state of the environment
        action (int): the previous action that was taken
        empty_obs (TigerObservation): an empty observation that needs to be filled and returned

        obs (TigerObservation): observation of the agent

        if state.tiger_location == TigerActions.LEFT:
            obs = random.choice([TigerObservationEnum.HEAR_LEFT, TigerObservationEnum.NONE])
        elif state.tiger_location == TigerActions.RIGHT:
            obs = random.choice([TigerObservationEnum.HEAR_RIGHT, TigerObservationEnum.NONE])
    else:
        # When opening a door, the agent does not hear anything
        obs = TigerObservationEnum.NONE

    return TigerObservation(obs=obs.value)
