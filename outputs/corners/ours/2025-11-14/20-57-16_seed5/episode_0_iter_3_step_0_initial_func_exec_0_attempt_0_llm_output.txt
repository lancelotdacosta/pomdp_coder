def initial_func(empty_state: MinigridState) -> MinigridState:
    """
    Input:
        empty_state (MinigridState): An empty state with only the walls filled into the grid
    Returns:
        state (MinigridState): the initial state of the environment
    """
    # Get the open cells in the grid
    open_cells = empty_state.get_type_indices(ObjectTypes.empty)
    
    if not open_cells:
        raise ValueError("No open cells available for placing the agent and goal.")

    # Define the center of the grid
    center_x, center_y = empty_state.width // 2, empty_state.height // 2

    # Create a Gaussian distribution centered at the grid's midpoint
    def gaussian(x, y, center_x, center_y, sigma=1.0):
        return np.exp(-((x - center_x) ** 2 + (y - center_y) ** 2) / (2 * sigma**2))

    # Compute the probabilities for each open cell
    probabilities = [
        gaussian(cell[0], cell[1], center_x, center_y) for cell in open_cells
    ]
    probabilities = np.array(probabilities)
    probabilities /= probabilities.sum()  # Normalize to sum to 1

    # Randomly select a position for the agent from the open cells with weighted probabilities
    agent_pos = tuple(open_cells[np.random.choice(len(open_cells), p=probabilities)])

    agent_dir = np.random.randint(0, 4)
    
    # Remove the agent's position from the list of open cells
    remaining_open_cells = [cell for cell in open_cells if cell != agent_pos]
    if not remaining_open_cells:
        raise ValueError("No open cells available for placing the goal.")
    
    # Compute the probabilities for the remaining open cells
    probabilities = [
        gaussian(cell[0], cell[1], center_x, center_y) for cell in remaining_open_cells
    ]
    probabilities = np.array(probabilities)
    probabilities /= probabilities.sum()  # Normalize to sum to 1

    # Randomly select a position for the goal from the remaining open cells with weighted probabilities
    goal_pos = tuple(remaining_open_cells[np.random.choice(len(remaining_open_cells), p=probabilities)])
    
    # Place the goal in the grid
    empty_state.grid[goal_pos] = ObjectTypes.goal
    
    # Create the initial state
    initial_state = MinigridState(
        grid=empty_state.grid.copy(),
        agent_pos=agent_pos,
        agent_dir=agent_dir,
        carrying=None,
    )
    
    return initial_state
