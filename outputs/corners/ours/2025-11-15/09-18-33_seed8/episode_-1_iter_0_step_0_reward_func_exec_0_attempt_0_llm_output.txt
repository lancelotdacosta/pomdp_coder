def reward_func(state, action, next_state):
    """
    Args:
        state (MinigridState): the state of the environment
        action (int): the action to be executed
        next_state (MinigridState): the next state of the environment
    Returns:
        reward (float): the reward of that state
        done (bool): whether the episode is done
    """
    
    def is_goal_reached(state):
        goal_positions = state.get_type_indices(ObjectTypes.goal)
        return next_state.agent_pos in goal_positions

    def is_wall_in_front(state):
        front_pos = state.front_pos
        return front_pos[0] < 0 or front_pos[0] >= state.width or front_pos[1] < 0 or front_pos[1] >= state.height or state.grid[front_pos] == ObjectTypes.wall

    # Check if the agent has reached the goal
    if is_goal_reached(next_state):
        return 1, True

    # Check if the action was moving forward into a wall
    if action == Actions.forward and is_wall_in_front(state):
        return -1, False

    # No reward for other actions
    return 0, False
