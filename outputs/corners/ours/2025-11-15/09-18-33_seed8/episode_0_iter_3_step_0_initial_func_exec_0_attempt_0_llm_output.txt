```python
def initial_func(empty_state: MinigridState):
    """
    Input:
        empty_state (MinigridState): An empty state with only the walls filled into the grid
    Returns:
        state (MinigridState): the initial state of the environment
    """
    
    def random_position(grid):
        """Generate a random position that is not a wall or the goal."""
        while True:
            x = np.random.randint(1, grid.shape[0] - 1)
            y = np.random.randint(1, grid.shape[1] - 1)
            if grid[x, y] == ObjectTypes.empty:
                return (x, y)
    
    def place_goal(grid):
        """Place a goal cell at a random position that is not a wall or the agent's position."""
        while True:
            x = np.random.randint(1, grid.shape[0] - 1)
            y = np.random.randint(1, grid.shape[1] - 1)
            if grid[x, y] == ObjectTypes.empty:
                grid[x, y] = ObjectTypes.goal
                return (x, y)

    # Get a copy of the empty state to modify
    state = MinigridState(
        grid=empty_state.grid.copy(),
        agent_pos=(0, 0),
        agent_dir=0,
        carrying=None
    )

    # Randomly place the agent
    state.agent_pos = random_position(state.grid)
    
    # Randomly set the agent's direction
    state.agent_dir = np.random.choice(list(Direction))
    
    # Place a goal cell
    place_goal(state.grid)
    
    return state
```

This function ensures that the agent is placed in a valid position, the agent's direction is randomly set, and a goal cell is placed in a valid position. The function returns the initial state of the environment.
