def reward_func(state, action, next_state):
    """
    Args:
        state (RockSampleState): the state of the environment
        action (int): the action to be executed
        next_state (RockSampleState): the next state of the environment
    Returns:
        reward (float): the reward of that state
        done (bool): whether the episode is done
    """
    # Define the reward for each type of action
    if action == RockSampleActions.SAMPLE:
        # If the agent samples a rock, check if it's good or bad
        rock_idx = state.at_rock()
        if rock_idx is not None:
            # If the agent is at a rock and samples it
            if state.rocks[rock_idx]:
                # Good rock
                reward = 10.0
            else:
                # Bad rock
                reward = -10.0
            return reward, True  # Episode ends after sampling a rock
        else:
            # Agent tried to sample but was not at a rock
            reward = -1.0
            return reward, False

    elif action == RockSampleActions.EXIT:
        # If the agent exits the environment
        reward = 10.0
        return reward, True  # Episode ends after exiting

    elif action in CHECK_ACTIONS:
        # If the agent checks a rock, no immediate reward
        rock_idx = state.at_rock()
        if rock_idx is not None:
            # If the agent is at a rock and checks it
            reward = 0.0
            return reward, False
        else:
            # Agent tried to check but was not at a rock
            reward = -1.0
            return reward, False

    else:
        # For all other actions (move), no immediate reward
        reward = 0.0
        return reward, False
