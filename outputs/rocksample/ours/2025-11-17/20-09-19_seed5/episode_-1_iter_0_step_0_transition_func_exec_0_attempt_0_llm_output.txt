    """
    Args:
        state (RockSampleState): the state of the environment 
        action (int): action to be taken in state `state`
    Returns:
        new_state (RockSampleState): the new state of the environment
    """

    if action == RockSampleActions.MOVE_NORTH:

            new_state.y += 1

        if new_state.y > 0:
            new_state.y -= 1
    elif action == RockSampleActions.MOVE_EAST:
        if new_state.x < GRID_SIZE - 1:
            new_state.x += 1
    elif action == RockSampleActions.MOVE_WEST:
        if new_state.x > 0:
            new_state.x -= 1
    elif action == RockSampleActions.SAMPLE:
        rock_idx = new_state.at_rock()
        if rock_idx is not None and rock_idx < NUM_ROCKS:
            # Sample the rock (this action does not change the state)
            pass
    elif action == RockSampleActions.EXIT:
        # Exit the environment (this action does not change the state)
        pass
    elif action in CHECK_ACTIONS:
        rock_idx = new_state.at_rock()
        if rock_idx is not None and rock_idx < NUM_ROCKS:
            # Check the rock (this action does not change the state)
            pass
    
    return new_state
