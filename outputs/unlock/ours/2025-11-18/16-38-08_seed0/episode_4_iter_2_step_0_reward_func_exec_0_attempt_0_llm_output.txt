def reward_func(state, action, next_state):
    """
    Args:
        state (MinigridState): the state of the environment
        action (int): the action to be executed
        next_state (MinigridState): the next state of the environment
    Returns:
        reward (float): the reward of that state
        done (bool): whether the episode is done
    """
    
    # Check if the agent has reached the goal
    if next_state.grid[next_state.agent_pos] == ObjectTypes.goal:
        return 1.0, True
    
    # Check if the agent has picked up a key
    if state.carrying is None and next_state.carrying == ObjectTypes.key:
        return 0.5, False
    
    # Check if the agent has unlocked a door
    if state.carrying == ObjectTypes.key and next_state.carrying is None:
        front_pos = next_state.front_pos
        if next_state.grid[front_pos] == ObjectTypes.open_door:
            return 0.2, False
    
    # Check if the agent is still in the same position (e.g., hitting a wall)
    if state.agent_pos == next_state.agent_pos:
        return -0.1, False
    
    # Default reward for other actions
    return 0.0, False
